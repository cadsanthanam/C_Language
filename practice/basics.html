Problem 1: Personal Information

üìù Task

Write a program that asks for and displays the following information:

  • Name (string)
  • Age (integer)
  • Height in meters (float)
  • Grade (character)
üí° Solution 
#include <stdio.h>

int main() {
    char name[50];
    int age;
    float height;
    char grade;
    
    printf("Enter your name: ");
    scanf("%s", name);
    
    printf("Enter your age: ");
    scanf("%d", &age);
    
    printf("Enter your height (meters): ");
    scanf("%f", &height);
    
    printf("Enter your grade: ");
    scanf(" %c", &grade);
    
    printf("\n--- Your Information ---\n");
    printf("Name: %s\n", name);
    printf("Age: %d years\n", age);
    printf("Height: %.2f meters\n", height);
    printf("Grade: %c\n", grade);
    
    return 0;
}

Problem 2: Simple Calculator

üìù Task

Create a program that takes two numbers and performs basic arithmetic operations.

üí° Solution 
#include <stdio.h>

int main() {
    float num1, num2;
    
    printf("Enter first number: ");
    scanf("%f", &num1);
    
    printf("Enter second number: ");
    scanf("%f", &num2);
    
    printf("\n--- Results ---\n");
    printf("%.2f + %.2f = %.2f\n", num1, num2, num1 + num2);
    printf("%.2f - %.2f = %.2f\n", num1, num2, num1 - num2);
    printf("%.2f * %.2f = %.2f\n", num1, num2, num1 * num2);
    
    if (num2 != 0) {
        printf("%.2f / %.2f = %.2f\n", num1, num2, num1 / num2);
    } else {
        printf("Cannot divide by zero!\n");
    }
    
    return 0;
}

Problem 3: Area and Perimeter

}

Problem 5: Simple Interest Calculator

üìù Task

Calculate simple interest and final amount.

Formula: SI = (P √ó R √ó T) / 100

Where P = Principal, R = Rate, T = Time

üí° Solution 
#include <stdio.h>

int main() {
    float principal, rate, time, interest, amount;
    
    printf("Enter principal amount: $");
    scanf("%f", &principal);
    
    printf("Enter rate of interest (%%): ");
    scanf("%f", &rate);
    
    printf("Enter time period (years): ");
    scanf("%f", &time);
    
    interest = (principal * rate * time) / 100;
    amount = principal + interest;
    
    printf("\n--- Loan Details ---\n");
    printf("Principal Amount: $%.2f\n", principal);
    printf("Rate of Interest: %.2f%%\n", rate);
    printf("Time Period: %.1f years\n", time);
    printf("Simple Interest: $%.2f\n", interest);
    printf("Total Amount: $%.2f\n", amount);
    
    return 0;
}

Challenging Problems

Problem 6: Swap Without Third Variable

üìù Task

Write a program to swap two numbers without using a third variable. Show both arithmetic and XOR methods.

üí° Solution 
#include <stdio.h>

int main() {
    int a, b;
    
    printf("Enter two numbers: ");
    scanf("%d %d", &a, &b);
    
    printf("Before swapping: a = %d, b = %d\n", a, b);
    
    // Method 1: Arithmetic approach
    a = a + b;
    b = a - b;
    a = a - b;
    
    printf("After swapping: a = %d, b = %d\n", a, b);
    
    // Method 2: XOR approach (reset first)
    a = a + b;
    b = a - b;
    a = a - b;  // Back to original
    
    printf("Original again: a = %d, b = %d\n", a, b);
    
    // XOR method
    a = a ^ b;
    b = a ^ b;
    a = a ^ b;
    
    printf("After XOR swap: a = %d, b = %d\n", a, b);
    
    return 0;
}

Problem 7: Character Analysis

üìù Task

Input a character and display its ASCII value and category (digit, uppercase, lowercase, special).

üí° Solution 
#include <stdio.h>

int main() {
    char ch;
    
    printf("Enter a character: ");
    scanf("%c", &ch);
    
    printf("Character: %c\n", ch);
    printf("ASCII Value: %d\n", ch);
    
    // Determine category
    if (ch >= '0' && ch <= '9') {
        printf("Category: Digit\n");
    } else if (ch >= 'A' && ch <= 'Z') {
        printf("Category: Uppercase Letter\n");
    } else if (ch >= 'a' && ch <= 'z') {
        printf("Category: Lowercase Letter\n");
    } else if (ch == ' ') {
        printf("Category: Space\n");
    } else {
        printf("Category: Special Character\n");
    }
    
    return 0;
}

Debugging Practice

Find and fix the errors in these code snippets:

Debug 1: Format Specifier Error

// What's wrong with this code?
#include <stdio.h>

int main() {
    float price = 19.99;
    printf("Price: %d dollars\n", price);
    return 0;
}
üí° Answer

Error: Using %d for a float variable.
Fix: Use %f instead of %d
printf("Price: %.2f dollars\n", price);

Debug 2: Missing Address Operator

// What's wrong with this code?
#include <stdio.h>

int main() {
    int age;
    printf("Enter your age: ");
    scanf("%d", age);
    printf("You are %d years old\n", age);
    return 0;
}
üí° Answer

Error: Missing & operator in scanf.
Fix: scanf("%d", &age);
scanf needs the address of the variable to store input.
